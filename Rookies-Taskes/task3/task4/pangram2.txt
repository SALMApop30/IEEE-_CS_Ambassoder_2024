#include <iostream>
#include <cctype> // for tolower
using namespace std;

int main() {
    int n;
    cin >> n;
    
    string s;
    cin >> s;
    
    bool letters[26] = {false};
    
    for (int i = 0; i < n; ++i) {
        char c = tolower(s[i]);
        letters[c - 'a'] = true;
    }
    
    bool is_pangram = true;
    for (int i = 0; i < 26; ++i) {
        if (!letters[i]) {
            is_pangram = false;
            break;
        }
    }
    
    if (is_pangram) {
        cout << "YES" << endl;
    } else {
        cout << "NO" << endl;
    }
    
    return 0;
}
